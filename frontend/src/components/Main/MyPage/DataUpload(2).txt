import React, { useState } from 'react';
import axios from 'axios';
import ElectricStats from "../ElectricStats"; // 그래프 컴포넌트

const DataUpload = () => {
  const [loading, setLoading] = useState(false);
  const [analysisData, setAnalysisData] = useState(null); // FastAPI 결과 저장

  const FASTAPI_URL = "http://43.201.202.195:8000/predict";

  const handleFileUpload = async (event) => {
    const file = event.target.files[0];
    if (!file) return;

    const formData = new FormData();
    formData.append('file', file);

    setLoading(true);
    try {
      const response = await axios.post(FASTAPI_URL, formData, {
        headers: { 'Content-Type': 'multipart/form-data' },
      });

      console.log("수신 데이터:", response.data);
      
      // ✅ 서버에서 온 'lastMonthAvg', 'nextMonthPred' 데이터를 저장
      setAnalysisData(response.data); 
      alert("한 달 단위 전력 분석이 완료되었습니다!");
    } catch (err) {
      console.error(err);
      alert("분석 중 오류가 발생했습니다. CSV 데이터가 충분한지(30일치 이상) 확인해주세요.");
    } finally {
      setLoading(false);
    }
  };

  return (
    <div style={{ padding: '20px' }}>
      <h2>⚡ WattMate 월간 전력 분석 및 예측</h2>
      
      <div style={{ marginBottom: '30px', padding: '20px', border: '1px solid #ddd', borderRadius: '10px' }}>
        <p>최근 1개월 데이터가 포함된 CSV 파일을 업로드해주세요.</p>
        <input type="file" onChange={handleFileUpload} accept=".csv" />
        {loading && <p>AI 모델이 월간 패턴을 분석 중입니다... 잠시만 기다려주세요.</p>}
      </div>

      {/* 분석 결과 데이터가 있을 때만 그래프 컴포넌트를 렌더링 */}
      {analysisData && (
        <ElectricStats 
          // ✅ 자식 컴포넌트(ElectricStats)가 기존 props 명칭(thisWeek 등)을 
          // 그대로 사용한다면 아래와 같이 매핑해줍니다.
          thisWeek={analysisData.lastMonthAvg} // 전달되는 값은 '한 달 평균'
          nextWeek={analysisData.nextMonthPred} // 전달되는 값은 '다음 달 예측'
        />
      )}
    </div>
  );
};

export default DataUpload;
